    DFT

    {atan2, cos, sin, sqrt} = Math
τ = 2 * Math.PI

DFT = (series) ->
  N = series.length
  rootN = sqrt(N)
  divRootN = (x) -> x / rootN

  [0...N].map (k) ->
    series.map ([x, y], n) ->
      theta = -τ * k * n / N
      [
        x *  cos(theta) + y *  sin(theta)
        x * -sin(theta) + y * -cos(theta)
      ]
    .reduce((a, b) ->
      [a[0] + b[0], a[1] + b[1]]
    , [0, 0])
    .map divRootN

inverseDFT = (series) ->
  do (x = DFT(series)) ->
    x.push(x.shift())
    x.reverse()

polar = (series) ->
  series.map ([r, i]) ->
    [
      sqrt(r * r + i * i)
      atan2(i, r) / τ
    ]

    Format for output

    format = (series) ->
  series.map (array) ->
    array.map (i) ->
      v = i.toFixed(3)
      "#{whitespace(10 - v.length)}#{v}"
    .join("")
  .join("\n")

whitespace = (n) ->
  [0...n].map(-> " ").join("")
